배열 arr[]과 위치 s, t가 있을 때,
arr[s], arr[s+1], … , arr[t-1]을 오른쪽으로 한 칸씩 이동하고,
arr[t]는 arr[s]로 복사하는 것을 ’1만큼 오른쪽으로 회전시켰다’고 한다.
예를 들어 길이가 8인 배열에서 s=2, t=6이면 다음 그림처럼 바뀐다.
길이가 n인 배열의 위치는 0, 1, 2, … , n-1이다.
문제 :
k를 인자로 받아서 k만큼 오른쪽으로 회전시키는 함수를 작성하라.
단, 1만큼 오른쪽으로 이동시키는 과정을 k번 반복해서는 안 된다.

인사이트 알고리즘 코딩 이벤트 사이트를 우연히 발견해서 나도 한번 풀어봤다.. 다양한 언어로 답을 제출하는거 같은데, 그중에 정답 가려내기도 쉽지 않을듯...



#include "stdafx.h"
#include <iostream>

using namespace std;

int * gnArr;
int gnTotalAryCnt;

void rotateAry(int s, int t, int k)
{
    int capacity=0, i=0, iSum=0 ;    
    int capacitySum = (t-s)+1;    
    int* nArrSum = new int[capacitySum];

    capacity = k;
    for( i=0; i<capacity; i++)
    {
        nArrSum[iSum] = gnArr[(t-k)+i];
        iSum++;
    }

    capacity = (t-s-k)+1;
    for( i=0; i<capacity; i++)
    {
        nArrSum[iSum] = gnArr[(s-1)+i];
        iSum++;
    } 
    
    for( i=0; i<capacitySum; i++)
    {
        gnArr[(s-1)+i] = nArrSum[ i ] ;        
    }
    
    delete [] nArrSum;            
}

int _tmain(int argc, _TCHAR* argv[])
{
    int s,t,k,i;    
    cout << "input Total Ary Cnt: " ; cin >> gnTotalAryCnt ;
    cout << "input s: " ; cin >> s ;
    cout << "input t: " ; cin >> t ;
    cout << "input k: " ; cin >> k ;
    
    gnArr = new int[gnTotalAryCnt];    
    
    for(i=0; i < gnTotalAryCnt; i++)
    {
        gnArr[i] = i+1;  //init array      
    }
    
    int nRotate= k % ((t-s)+1);
    cout << "rotate times: " << nRotate << endl;

    if( nRotate ) 
    {        
        rotateAry ( s, t, k);
    } else  {
        cout << "Doesn't need to rotate!" << endl;
    }

    cout << "\n* Result: ";
    for(i=0; i < gnTotalAryCnt; i++)
    {
        cout << " " << gnArr[i] ;
    }
    cout << endl;

    delete [] gnArr;
    
    return 0;
}
