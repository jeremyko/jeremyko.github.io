boost - regex

정규표현식 관련된 방대한 라이브러리중에서 boost::regex 와 boost::smatch 위주로 설명
전자는 정규표현식을 정의하기 위해서, 후자는 조회결과를 저장하기 위한 목적.


#include <boost/regex.hpp>
#include <locale>
#include <iostream>

int main()
{
std::locale::global(std::locale("German"));
   std::string s = "Boris Schling";
   boost::regex expr("\\w+\\s\\w+");
   std::cout << boost::regex_match(s, expr) << std::endl;
}

boost::regex_match 는 전체문자열과 일치시에 true를 리턴.
문자열 검색을 위해서는 boost::regex_search() 를 사용

#include <boost/regex.hpp>
#include <locale>
#include <iostream>

int main()
{    
    std::string s = "1234em KKKK";

    boost::regex expr("(\\w+)\\s(\\w+)");

    boost::smatch what;

    if (boost::regex_search(s, what, expr))
    {
        std::cout << what[0] << std::endl;
        std::cout << what[1] << std::endl;
        std::cout << what[2] << std::endl;      
    }
}

boost::regex_replace()

#include <boost/regex.hpp>
#include <locale>
#include <iostream>

int main()
{
   std::locale::global(std::locale("German"));
   std::string s = " Test String ";
   boost::regex expr("\\s");
   std::string fmt("_");
  
   //전체 문자열에 대해서 치환이 이루어진다.
   std::cout << boost::regex_replace(s, expr, fmt) << std::endl; //_Test_String_
}

치환되어지는 format문자열은 그룹핑 정규식으로 조회된 sub-string에 접근할수 있다.
#include <boost/regex.hpp>
#include <locale>
#include <iostream>

int main()
{  
    std::string s = "111 222";
    boost::regex expr("(\\w+)\\s(\\w+)");
    std::string fmt("\\2 \\1");
    std::cout << boost::regex_replace(s, expr, fmt) << std::endl; // 222 111 출력
}

그런데 정말로 "\2 \1" 로 지환하고 싶다면 boost::regex_constants::format_literal
을 4번째 인자로 넘기면 된다.

#include <boost/regex.hpp>
#include <locale>
#include <iostream>

int main()
{
   std::locale::global(std::locale("German"));
   std::string s = "Boris Schäling";
   boost::regex expr("(\\w+)\\s(\\w+)");
   std::string fmt("\\2 \\1");
   std::cout << boost::regex_replace(s, expr, fmt,
   boost::regex_constants::format_literal) << std::endl;
   /* \2 \1 출력 */
}
