boost::strand

하나의 I/O Service 에 대해 여러개의 쓰레드에서 run()함수를 호출하는 경우,
이미 알아본것처럼 핸들러 함수가 동시에 실행이 된다. 즉,


#include <iostream>
#include <boost/asio.hpp>
#include <boost/thread.hpp>
#include <boost/bind.hpp>
#include <boost/date_time/posix_time/posix_time.hpp>

class printer
{
public:
    printer(boost::asio::io_service& io) :
                    timer1_(io, boost::posix_time::seconds(1)),
                    timer2_(io, boost::posix_time::seconds(1)),
                    count_(0)
    {
timer1_.async_wait(boost::bind(&printer::print1, this));
        timer2_.async_wait(boost::bind(&printer::print2, this));
    }

    ~printer()
    {
     std::cout << "Final count is " << count_ << "\n";
    }

    void print1()
    {
     if (count_ < 10)
     {
         std::cout << "Timer 1: " << count_ << "\n";
            ++count_;

            timer1_.expires_at(timer1_.expires_at() + boost::posix_time::seconds(1));
            timer1_.async_wait(boost::bind(&printer::print1, this));
        }
    }

    void print2()
    {
     if (count_ < 10)
     {
         std::cout << "Timer 2: " << count_ << "\n";
            ++count_;

            timer2_.expires_at(timer2_.expires_at() + boost::posix_time::seconds(1));
            timer2_.async_wait(boost::bind(&printer::print2, this));
        }
    }

    private:
    boost::asio::deadline_timer timer1_;
    boost::asio::deadline_timer timer2_;
    int count_;
};

int main()
{
   boost::asio::io_service io;
   printer p(io);
   boost::thread t(boost::bind(&boost::asio::io_service::run, &io));
io.run();
t.join();

   return 0;
}

출력:

Timer 2: Timer 1: 00

Timer 2: Timer 1: 22

Timer 1: Timer 2: 44

Timer 2: Timer 1: 66

Timer 1: Timer 2: 88

Final count is 10

핸들러 함수 print1, print2 가 동기처리 없이 동시에 수행이 되기 때문에, 출력이 이상하다.
이러한 핸들러 함수간의 동기화 문제를 다루기 위해서 boost::strand 클래스가 있다.
io_service::strand::wrap()은 새로운 핸들러를 생성하고, 핸들러가 자신을 통해서만
수행되게끔 해준다. 이를통해 run()함수가 서로 다른 스레드에서 호출되더라도
핸들러 함수가 동시에 실행되는것을 방지한다. 위의 예제를 수정해보면,

#include <iostream>
#include <boost/asio.hpp>
#include <boost/thread.hpp>
#include <boost/bind.hpp>
#include <boost/date_time/posix_time/posix_time.hpp>

class printer
{
    public:
    printer(boost::asio::io_service& io) : strand_(io),
                    timer1_(io, boost::posix_time::seconds(1)),
                    timer2_(io, boost::posix_time::seconds(1)),
                    count_(0)
    {
timer1_.async_wait(strand_.wrap(boost::bind(&printer::print1, this)));
        timer2_.async_wait(strand_.wrap(boost::bind(&printer::print2, this)));
    }

    ~printer()
    {
     std::cout << "Final count is " << count_ << "\n";
    }

    void print1()
    {
     if (count_ < 10)
     {
            std::cout << "Timer 1: " << count_ << "\n";
            ++count_;

            timer1_.expires_at(timer1_.expires_at() + boost::posix_time::seconds(1));
            timer1_.async_wait(strand_.wrap(boost::bind(&printer::print1, this)));
        }
    }

    void print2()
    {
     if (count_ < 10)
     {
            std::cout << "Timer 2: " << count_ << "\n";
            ++count_;

            timer2_.expires_at(timer2_.expires_at() + boost::posix_time::seconds(1));
            timer2_.async_wait(strand_.wrap(boost::bind(&printer::print2, this)));
        }
    }
    private:
    boost::asio::strand strand_;
    boost::asio::deadline_timer timer1_;
    boost::asio::deadline_timer timer2_;
    int count_;
};

int main()
{
   boost::asio::io_service io;
   printer p(io);
   boost::thread t(boost::bind(&boost::asio::io_service::run, &io));
io.run();
t.join();

   return 0;
}
